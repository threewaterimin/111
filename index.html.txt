      });
      // 显示目标页面并添加动画
      const target = document.getElementById('page-' + pageId);
      if (target) {
        target.classList.remove('hidden');
        target.classList.add('page-enter');
      }
    }

    // --- 角色编辑逻辑 ---
    function openCharEditor(id = null) {
      editingCharId = id;
      const title = document.getElementById('char-page-title');
      const btn = document.getElementById('char-save-btn');
      const nameInput = document.getElementById('char-input-name');
      const descInput = document.getElementById('char-input-desc');
      const avatar = document.getElementById('char-avatar-preview');

      if (id) {
        const char = CHARACTERS.find(c => c.id === id);
        title.innerText = 'Edit Character';
        btn.innerText = 'Save Changes';
        nameInput.value = char.name;
        descInput.value = char.desc;
        avatar.className = `w-24 h-24 rounded-full border-2 border-dashed border-pink-300 flex items-center justify-center shadow-inner ${char.color} ${char.iconColor}`;
        avatar.innerHTML = `<i data-lucide="smile" class="w-12 h-12"></i>`;
      } else {
        title.innerText = 'Create Character';
        btn.innerText = 'Create Character';
        nameInput.value = '';
        descInput.value = '';
        avatar.className = `w-24 h-24 rounded-full border-2 border-dashed border-pink-300 flex items-center justify-center shadow-inner bg-pink-100 text-pink-400`;
        avatar.innerHTML = `<i data-lucide="camera" class="w-8 h-8"></i>`;
      }
      lucide.createIcons();
      nav('create-char');
    }

    function saveCharacter() {
      const name = document.getElementById('char-input-name').value || 'Unnamed';
      const desc = document.getElementById('char-input-desc').value || 'No description';
      
      if (editingCharId) {
        const idx = CHARACTERS.findIndex(c => c.id === editingCharId);
        CHARACTERS[idx].name = name;
        CHARACTERS[idx].desc = desc;
      } else {
        CHARACTERS.push({
          id: Date.now().toString(),
          name, desc,
          color: 'bg-fuchsia-100', iconColor: 'text-fuchsia-500'
        });
      }
      renderCharacters();
      lucide.createIcons();
      nav('characters');
    }
